\begin{frame}[t]
 \frametitle{Fundamental Symmetries in Physics}
 \begin{block}{Discrete transformations}
  \begin{itemize}
   \item<+-> \alert{Parity transformation $\mathcal{P}$}: transformation to the mirror image
   \item<+-> \alert{Charge conjugation $\mathcal{C}$}: exchange of particle/antiparticle
   \item<+-> \alert{Time reversal $\mathcal{T}$}: reversal of the direction of time
  \end{itemize}
 \end{block}
 \only<1>{
 \begin{block}{Symmetry under parity transformation}
  \begin{itemize}
   \item Nature works also in its mirror image, mostly\ldots
   \item Chemical chirality: left-handed and right-handed sugars
   \item \alert{Parity violation}: $W$ and $Z$ bosons only interact with left-handed particles, not with right-handed particles
   \item Some supersymmetric particles also violate parity symmetry
   \item At \alert{Jefferson Lab}: parity violation as a probe
  \end{itemize}
 \end{block}
 }
 \only<2>{
 \begin{block}{Symmetry under charge conjugation}
  \begin{itemize}
   \item Two opposite electrical charges attract each other
   \item Transform each of them and they will still attract
   \item Charge symmetry violation: $W$ and $Z$ bosons only interact with left-handed neutrinos, not with left-handed antineutrinos
  \end{itemize}
 \end{block}
 }
 \only<3>{
 \begin{block}{Symmetry under time reversal}
  \begin{itemize}
   \item A movie of billiard balls rolling on a table could be played backwards, and you wouldn't notice the difference
   \item Disclaimer: eggs don't `unbreak'; google `arrow of time' to find out what physics has to say about this
  \end{itemize}
 \end{block}
 }
 \only<4>{
 \begin{block}{Symmetry under $\mathcal{C}\mathcal{P}\mathcal{T}$ transformation}
  \begin{itemize}
   \item $\mathcal{C}\mathcal{P}\mathcal{T}$-symmetry must be satisfied for very fundamental reasons
   \item But $\mathcal{C}\mathcal{P}$-symmetry is violated a tiny bit!
   \item Does that mean that $\mathcal{T}$-symmetry is violated?
   \item That's how it would seem\ldots
  \end{itemize}
 \end{block}
 }
\end{frame}
